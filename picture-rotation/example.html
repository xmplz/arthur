Пример на JavaScript

let jsRotation = 0;
const jsRotationStep = -15;
document.getElementById('').addEventListener("click", function () {
     jsRotation += jsRotationStep;
     document.getElementById('').style.transform = `rotate(${jsRotation}deg)`;
});

Пример на JQuery

let jqRotation = 0;
const jqRotationStep = 15;
$(`#`).click(function () {
     jqRotation += jqRotationStep;
     $(`#`).css("transform", `rotate(${jqRotation}deg)`);
});
